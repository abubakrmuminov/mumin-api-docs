## ğŸš€ Quick Start

```kotlin
import ink.mumin.sdk.MuminClient
import kotlin.time.Duration.Companion.minutes

val client = MuminClient {
    apiKey = "YOUR_API_KEY"
    cache {
        enabled = true
        ttl = 30.minutes
    }
}

suspend fun main() {
    val hadith = client.hadiths.random()
    println("Hadith: ${hadith.translation?.text}")
}
```
